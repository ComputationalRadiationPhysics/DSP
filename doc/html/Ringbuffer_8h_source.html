125 int Ringbuffer<Type>::writeFromHost(Type &inputOnHost)
126 {
127  sem_wait(&space); // is there space in buffer?
128  sem_wait(&mtx); // lock buffer
129 
130  buffer.at(head) = inputOnHost;
131  head = ++head % bufferSize; // move head
132 
133  sem_post(&mtx); // unlock buffer
134  sem_post(&usage); // tell them that there is something in buffer
135 
136  return 0;
