314 {
315  int i;
316  float agiant, s1, s2, s3, xabs, x1max, x3max, temp;
317 
318  s1 = 0;
319  s2 = 0;
320  s3 = 0;
321  x1max = 0;
322  x3max = 0;
323  agiant = LM_SQRT_GIANT / ((float) n);
324 
325  //sum squares
326  for (i = 0; i < n; i++) {
327  xabs = fabs(x[i]);
328  if (xabs > LM_SQRT_DWARF && xabs < agiant) {
329  //sum for intermediate components
330  s2 += xabs * xabs;
331  continue;
332  }
333 
334  if (xabs > LM_SQRT_DWARF) {
335  //sum for large components
336  if (xabs > x1max) {
337  temp = x1max / xabs;
338  s1 = 1 + s1 * SQR(temp);
339  x1max = xabs;
340  } else {
341  temp = xabs / x1max;
342  s1 += SQR(temp);
343  }
344  continue;
345  }
346  //sum for small components
347  if (xabs > x3max) {
348  temp = x3max / xabs;
349  s3 = 1 + s3 * SQR(temp);
350  x3max = xabs;
351  } else {
352  if (xabs != 0.) {
353  temp = xabs / x3max;
354  s3 += SQR(temp);
355  }
356  }
357  }
358 
360 
361  if (s1 != 0)
362  *result = x1max * sqrt(s1 + (s2 / x1max) / x1max);
363  else if (s2 != 0) {
364  if (s2 >= x3max)
365  *result = sqrt(s2 * (1 + (x3max / s2) * (x3max * s3)));
366  else
367  *result = sqrt(x3max * ((s2 / x3max) + (x3max * s3)));
368  }
369  else
370  *result = x3max * sqrt(s3);
371 
372 }

605  rdiag[k] *= sqrt(temp);

610  }
611  }

615  }

707  //diag := norms of the columns of the initial jacobian
708  for (j = 0; j < n; j++) {
709  diag[j] = wa2[j];
710  if (wa2[j] == 0.)
711  diag[j] = 1.;
712  }
713  }
714  //use diag to scale x, then calculate the norm
715  for (j = 0; j < n; j++)
716  wa3[j] = diag[j] * x[j];
717  euclidNorm(n, wa3, &xnorm);
718  //initialize the step bound delta
719  delta = factor * xnorm;
720  if (delta == 0.)
721  delta = factor;
722  }
723 
724  //form (q transpose)*fvec and store first n components in qtf
725 
726  for (i = 0; i < m; i++)
727  wa4[i] = fvec[i];
728 
729  for (j = 0; j < n; j++) {
730  temp3 = fjac[j * m + j];
731  if (temp3 != 0.) {
732  sum = 0;
733  for (i = j; i < m; i++)
734  sum += fjac[j * m + i] * wa4[i];
735  temp = -sum / temp3;
736  for (i = j; i < m; i++)
737  wa4[i] += fjac[j * m + i] * temp;
738  }
739  fjac[j * m + j] = wa1[j];
740  qtf[j] = wa4[j];
741  }
742 
743  //compute norm of scaled gradient and test for convergence
744 
745  gnorm = 0;
746  if (fnorm != 0) {
747  for (j = 0; j < n; j++) {
748  if (wa2[ipvt[j]] == 0)
749  continue;
750 
751  sum = 0.;
752  for (i = 0; i <= j; i++)
753  sum += fjac[j * m + i] * qtf[i] / fnorm;
754  gnorm = MAX(gnorm, fabs(sum / wa2[ipvt[j]]));
755  }
756  }
757 
758  if (gnorm <= gtol) {
759  *info = 4;
760  return;
761  }
762 
763  //rescale if necessary
764 
765  if (mode != 2) {
766  for (j = 0; j < n; j++)
767  diag[j] = MAX(diag[j], wa2[j]);
768  }
769

771  //determine the levenberg-marquardt parameter
772 
773  lmpar(n, fjac, m, ipvt, diag, qtf, delta, &par,
774  wa1, wa2, wa3, wa4);
775 
776  //store the direction p and x + p; calculate the norm of p
777 
778  for (j = 0; j < n; j++) {
779  wa1[j] = -wa1[j];
780  wa2[j] = x[j] + wa1[j];
781  wa3[j] = diag[j] * wa1[j];
782  }
783  euclidNorm(n, wa3, &pnorm);
784 
785  //on the first iteration, adjust the initial step bound
786 
787  if (*nfev <= 1 + n)
788  delta = MIN(delta, pnorm);
789 
790  //evaluate the function at x + p and calculate its norm
791 
792  *info = 0;
793  evaluate<tex>(wa2, m, wa4, indexDataset, xOffset, xStep);
794  ++(*nfev);
795 
796  euclidNorm(m, wa4, &fnorm1);
797 
798  //compute the scaled actual reduction
799 
800  if (0.1 * fnorm1 < fnorm)
801  actred = 1 - SQR(fnorm1 / fnorm);
802  else
803  actred = -1;
804 
805  //compute the scaled predicted reduction and the scaled directional derivative
806 
807  for (j = 0; j < n; j++) {
808  wa3[j] = 0;
809  for (i = 0; i <= j; i++)
810  wa3[i] += fjac[j * m + i] * wa1[ipvt[j]];
811  }
812  euclidNorm(n, wa3, &temp1);
813  temp1 /= fnorm;
814  temp2 = sqrt(par) * pnorm / fnorm;
815  prered = SQR(temp1) + 2 * SQR(temp2);
816  dirder = -(SQR(temp1) + SQR(temp2));
817 
818  //compute the ratio of the actual to the predicted reduction
819 
820  ratio = prered != 0 ? actred / prered : 0;
821 
822  //update the step bound
823 
824  if (ratio <= 0.25) {
825  if (actred >= 0.)
826  temp = 0.5;
827  else
828  temp = 0.5 * dirder / (dirder + 0.55 * actred);
829  if (0.1 * fnorm1 >= fnorm || temp < 0.1)
830  temp = 0.1;
831  delta = temp * MIN(delta, pnorm / 0.1);
832  par /= temp;
833  } else if (par == 0. || ratio >= 0.75) {
834  delta = pnorm / 0.5;
835  par *= 0.5;
836  }
837 
838  //test for successful iteration
839 
840  if (ratio >= 0.0001) {
841  //yes, success: update x, fvec, and their norms
842  for (j = 0; j < n; j++) {
843  x[j] = wa2[j];
844  wa2[j] = diag[j] * x[j];
845  }
846  for (i = 0; i < m; i++)
847  fvec[i] = wa4[i];
848  euclidNorm(n, wa2, &xnorm);
849  fnorm = fnorm1;
850  iter++;
851  }
852 
853  //tests for convergence (otherwise *info = 1, 2, or 3)
854 
855  *info = 0; //do not terminate (unless overwritten by nonzero)
856  if (fabs(actred) <= ftol && prered <= ftol && 0.5 * ratio <= 1)
857  *info = 1;
858  if (delta <= xtol * xnorm)
859  *info += 2;
860  if (*info != 0)
861  return;
862 
863  //tests for termination and stringent tolerances
864 
865  if (*nfev >= maxfev)
866  *info = 5;
867  if (fabs(actred) <= LM_MACHEP &&
868  prered <= LM_MACHEP && 0.5 * ratio <= 1)
869  *info = 6;
870  if (delta <= LM_MACHEP * xnorm)
871  *info = 7;
872  if (gnorm <= LM_MACHEP)
873  *info = 8;
874  if (*info != 0)
875  return;
876 
877  //repeat if iteration unsuccessful
878  } while (ratio < 0.0001);
879  } while (1);
880 }
